[0-9]+ -- int literal

-[expression] -- negative expression

"[.*]" -- text literal

[a-z|A-Z]+ -- name

([name|,*]+) -- designation list

([expression|,*]+) -- tuple

\[[expression|,*]+\] -- array

function [designation list] [expression] -- function

[expression] [+|-|*|/|<|>|==|!=] [expression] -- binary expression

\.[name] -- symbol expression

[expression][expression] -- invocation expression

([expression]) -- parenthesized expression

using [expression list] as [designation list] [expression] -- using expression

[designation list | name] -> [expression] -- function arrow expression

[name]:[expression] -- key-value-pair
[name]=[expression] -- binding

([name|,*]+) -- binding list

\{[key-value-pair|,]+\} -- map expression
[key-value-pair] -- single element map

[binding list | binding] -> [expression] -- using arrow expression
